           $(document).ready(function () {
            
                function swapArray(valueA, valueB, valueC, valueD){
                    
        var valueA = prompt("Type the first character/number in your array?");
                
        var valueB = prompt("Type the second characte/number in your array?");
                    
        var valueC = prompt("Type the third character/number in your array?");
                   
        var valueD = prompt("Type the fourth character/number in your array?");
        
        var valueE = prompt("Type the fifth character/number in your array?");
                 
                    
                   var arr = [valueA, valueB, valueC, valueD, valueE];
                    //orignal unaltered array
                    
                    
                
                //iarr  is used in question 8 as 'inverse array' so sarr is used for 'swapped array'
                
                    var sarr = [valueA, valueB, valueC, valueD, valueE]; 
                    //swapped array variable
                    
                    //calling on iarr values [0, 1, 2, 3, 4]
                    
                    var temp; //variable to store a given value
                    temp = sarr[0]; // store valueA in temp variable
                    sarr[0] = sarr[4]; // valueA takes the value of valueE
                    sarr[4] = temp;  //valueE takes the value of temp, which is the stored value of valueA
                    //valueA and valueE have been swapped
                    
                     temp = sarr[1]; //valueB is stored in temp variable
                    sarr[1] = sarr[3]; //valueB takes the value of valueD
                sarr[3] = temp; //valueD takes the new value of temp, which is the stored value of valueB
                    //valueB and valueD have been swapped
                    
                    //valueC is uneffected because it is the middle value in the array
                    
               $("#q9-result").html("Original array: " + arr + "
" + "Swapped array: " + sarr);
                   
                  
                        
                
          }
                //ends reverseArray function
                

                $("#q9-button").on("click", function () {
                     swapArray();
                    
                
                //ends click function 
                });


                //ends document ready function
            });
